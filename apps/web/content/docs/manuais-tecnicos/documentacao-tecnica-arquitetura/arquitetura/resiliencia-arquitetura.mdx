## 1. Camadas de Error Handling

### 1.1 Tipagem de Erros

Estamos evoluindo para uma arquitetura de erros estruturados:

```typescript
// Base class (packages/api/src/lib/errors)
export class DomainError extends Error {
    constructor(
        public code: string,
        message: string,
        public statusCode: number = 400,
        public context?: Record<string, unknown>
    ) {
        super(message);
        this.name = this.constructor.name;
    }
}

// Específicos de domínio
export class ValidationError extends DomainError {
    constructor(message: string, details?: Record<string, string>) {
        super("VALIDATION_ERROR", message, 400, { fields: details });
    }
}

export class NotFoundError extends DomainError {
    constructor(resource: string, id?: string) {
        super(
            "NOT_FOUND",
            `${resource} not found${id ? `: ${id}` : ""}`,
            404
        );
    }
}

export class UnauthorizedError extends DomainError {
    constructor(message = "Unauthorized") {
        super("UNAUTHORIZED", message, 401);
    }
}

export class RateLimitError extends DomainError {
    constructor(public retryAfterMs: number) {
        super(
            "RATE_LIMIT_EXCEEDED",
            "Too many requests",
            429,
            { retryAfterMs }
        );
    }
}

export class IntegrationError extends DomainError {
    constructor(
        service: string,
        message: string,
        public originalError?: Error
    ) {
        super(
            "INTEGRATION_ERROR",
            `Failed to call ${service}: ${message}`,
            503,
            { service }
        );
    }
}
```

### 1.2 Tratamento em Server Actions

```typescript
// packages/api/src/lib/errors/action-error-handler.ts
import { DomainError, ValidationError, RateLimitError } from "@/lib/errors";

export type ActionResponse<T = void> = 
    | { success: true; data: T }
    | { success: false; error: string; code?: string; retryAfterMs?: number };

export function handleActionError(
    error: unknown,
    context?: { action?: string; userId?: string }
): ActionResponse {
    // Log estruturado ANTES de processamento
    logger.error("action.error", {
        action: context?.action,
        userId: context?.userId,
        errorType: error instanceof DomainError ? error.code : "UNKNOWN",
        message: error instanceof Error ? error.message : "Unknown error"
    });
    
    // Rate limit: cliente pode retry
    if (error instanceof RateLimitError) {
        return {
            success: false,
            code: "RATE_LIMIT_EXCEEDED",
            error: "Too many requests. Try again in a moment.",
            retryAfterMs: error.retryAfterMs
        };
    }
    
    // Validation: usuário precisa corrigir input
    if (error instanceof ValidationError) {
        return {
            success: false,
            code: "VALIDATION_ERROR",
            error: error.message
        };
    }
    
    // Domain errors: mensagens seguras
    if (error instanceof DomainError) {
        return {
            success: false,
            code: error.code,
            error: error.message
        };
    }
    
    // Fallback: erro desconhecido (não expor detalhes)
    return {
        success: false,
        error: "An unexpected error occurred. Please try again or contact support."
    };
}

// Uso em actions
export async function createTicketAction(
    input: CreateTicketInput
): Promise<ActionResponse<Ticket>> {
    try {
        const session = await getProtectedSession();
        const result = await ticketService.create(input, session.user.id);
        
        logger.info("ticket.created", {
            ticketId: result.id,
            userId: session.user.id
        });
        
        return { success: true, data: result };
    } catch (error) {
        return handleActionError(error, {
            action: "createTicket",
            userId: (await getProtectedSession()).user.id
        });
    }
}
```

```typescript
// packages/api/src/lib/trpc-error-handler.ts
import { TRPCError } from "@trpc/server";
import { DomainError, ValidationError, NotFoundError } from "./errors";

export function mapDomainErrorToTRPC(error: unknown): TRPCError {
    
    if (error instanceof ValidationError) {
        return new TRPCError({
            code: "BAD_REQUEST",
            message: error.message,
            cause: error
        });
    }
    
    if (error instanceof NotFoundError) {
        return new TRPCError({
            code: "NOT_FOUND",
            message: error.message
        });
    }
    
    if (error instanceof DomainError) {
        // Mapear status code para tRPC code
        const codeMap: Record<number, TRPCErrorCode> = {
            400: "BAD_REQUEST",
            401: "UNAUTHORIZED",
            403: "FORBIDDEN",
            404: "NOT_FOUND",
            429: "TOO_MANY_REQUESTS",
            503: "INTERNAL_SERVER_ERROR"
        };
        
        return new TRPCError({
            code: codeMap[error.statusCode] ?? "INTERNAL_SERVER_ERROR",
            message: error.message,
            cause: error
        });
    }
    
    return new TRPCError({
        code: "INTERNAL_SERVER_ERROR",
        message: "An unexpected error occurred"
    });
}

// Uso em procedimento tRPC
export const auditRouter = createRouter({
    list: defineQuery({
        auth: "authenticated",
        handler: async ({ ctx, input }) => {
            try {
                return await auditService.listFiltered(input);
            } catch (error) {
                throw mapDomainErrorToTRPC(error);
            }
        }
    })
});
```

## 2. Observabilidade: Logging Estruturado

### 2.1 Estratégia de Logging

Nossa abordagem usa **JSON estruturado** com namespace hierárquicos:

**Namespaces padrão:**
- `feature.event.phase` → `ticket.created.success`
- `system.component.action` → `auth.session.validated`

**Campos obrigatórios por tipo:**

| Campo | Tipo | Exemplo | Quando |
|-------|------|---------|--------|
| `namespace` | string | `ticket.created.success` | Sempre (parte da chave) |
| `timestamp` | ISO 8601 | `2026-03-29T14:30:45.123Z` | Sempre (adicionado pelo logger) |
| `userId` | string | UUID | Request com autenticação |
| `companyId` | string | UUID | Request com tenant |
| `durationMs` | number | 1250 | Operações I/O |
| `status` | string | `success` \| `error` | Fase final de processo |
| `error` | string | `VALIDATION_ERROR` | Em erros |
| `details` | object | `{ticketId: "..."}` | Metadata importante |

### 2.2 Implementação do Logger

```typescript
// packages/api/src/lib/observability/logger.ts
import pino from "pino";

const pinoLogger = pino({
    level: process.env.LOG_LEVEL ?? "info",
    transport:
        process.env.NODE_ENV === "development"
            ? { target: "pino-pretty" }
            : undefined,
    base: {
        service: "documentacao-portal",
        env: process.env.NODE_ENV,
        version: process.env.APP_VERSION
    }
});

export type LogContext = {
    userId?: string;
    companyId?: string;
    ticketId?: string;
    sessionId?: string;
    ip?: string;
    requestId?: string;
};

export class Logger {
    constructor(private context: LogContext = {}) {}
    
    private log(
        level: "debug" | "info" | "warn" | "error",
        namespace: string,
        data?: Record<string, unknown>
    ) {
        pinoLogger[level](
            {
                namespace,
                ...this.context,
                timestamp: new Date().toISOString(),
                ...data
            },
            `[${namespace}]`
        );
    }
    
    info(namespace: string, data?: Record<string, unknown>) {
        this.log("info", namespace, data);
    }
    
    error(namespace: string, error: Error | unknown, data?: Record<string, unknown>) {
        this.log("error", namespace, {
            ...data,
            error: error instanceof Error ? error.message : String(error),
            stack: error instanceof Error ? error.stack : undefined
        });
    }
    
    warn(namespace: string, data?: Record<string, unknown>) {
        this.log("warn", namespace, data);
    }
    
    debug(namespace: string, data?: Record<string, unknown>) {
        this.log("debug", namespace, data);
    }
    
    // Para operações timed
    async time<T>(
        namespace: string,
        fn: () => Promise<T>,
        data?: Record<string, unknown>
    ): Promise<T> {
        const start = performance.now();
        
        try {
            const result = await fn();
            const durationMs = Math.round(performance.now() - start);
            
            this.info(namespace, {
                status: "success",
                durationMs,
                ...data
            });
            
            return result;
        } catch (error) {
            const durationMs = Math.round(performance.now() - start);
            
            this.error(namespace, error, {
                status: "error",
                durationMs,
                ...data
            });
            
            throw error;
        }
    }
}

export const logger = new Logger();

// Com context
export function createLogger(ctx: LogContext) {
    return new Logger(ctx);
}
```

### 3.2 Implementação com Redis

```typescript
// src/lib/rate-limit/redis-limiter.ts
import { redis } from "@/lib/redis";

export class RateLimiter {
    async consume(
        scope: "user" | "ip" | "company",
        identifier: string,
        action: string
    ): Promise<{ allowed: boolean; retryAfterMs?: number }> {
        
        const config = this.getConfig(scope, action);
        if (!config) return { allowed: true }; // Sem limit
        
        const key = `ratelimit:${scope}:${identifier}:${action}`;
        const current = await redis.incr(key);
        
        // Primeira vez: set expiry
        if (current === 1) {
            await redis.expire(key, Math.ceil(config.windowMs / 1000));
        }
        
        // Verification
        const allowed = current <= config.maxAttempts;
        
        if (!allowed) {
            const ttl = await redis.ttl(key);
            return { 
                allowed: false, 
                retryAfterMs: ttl * 1000 
            };
        }
        
        return { allowed: true };
    }
    
    private getConfig(scope: string, action: string) {
        if (scope === "user") {
            return actionLimits[action as keyof typeof actionLimits];
        }
        if (scope === "ip") {
            return ipLimits;
        }
        if (scope === "company") {
            return companyQuotas["api:calls"];
        }
    }
}

export const limiter = new RateLimiter();
```

### 3.3 Uso em Actions

```typescript
export async function createTicketAction(input: CreateTicketInput) {
    const session = await getProtectedSession();
    const req = getRequest();
    
    // Check rate limit
    const userLimit = await limiter.consume("user", session.user.id, "user:create-ticket");

    }

    return { success: true, data: ticket };
}
```

## 4. Circuit Breaker para Integrações

### 4.1 Implementação de Circuit Breaker

```typescript
// src/lib/circuit-breaker/circuit-breaker.ts

export type CircuitState = "closed" | "open" | "half-open";

export interface CircuitBreakerConfig {
    failureThreshold: number;     // Quantas falhas antes de abrir
    failureWindow: number;        // Janela em ms para contar falhas
    cooldownMs: number;           // Tempo em half-open antes de tentar fechar
    timeout?: number;             // Timeout per call
}

export class CircuitBreaker<T> {
    private state: CircuitState = "closed";
    private failureCount = 0;
    private lastFailureTime: number | null = null;
    private nextAttemptTime: number | null = null;
    
    constructor(
        private name: string,
        private config: CircuitBreakerConfig,
        private fn: () => Promise<T>
    ) {}
    
    async execute(): Promise<T> {
        // Se aberto, verificar se tempo de cooldown passou
        if (this.state === "open") {
            if (Date.now() < (this.nextAttemptTime ?? 0)) {
                throw new CircuitBreakerOpenError(
                    `Circuit breaker  "${this.name}" is open`,
                    this.nextAttemptTime! - Date.now()
                );
            }
            
            // Tentar half-open
            this.state = "half-open";
        }
        
        try {
            const result = await this.executeWithTimeout();
            this.onSuccess();
            return result;
        } catch (error) {
            this.onFailure();
            throw error;
        }
    }
    
    private async executeWithTimeout(): Promise<T> {
        if (!this.config.timeout) {
            return this.fn();
        }
        
        return Promise.race([
            this.fn(),
            new Promise<never>((_, reject) =>
                setTimeout(
                    () => reject(new Error("Circuit breaker timeout")),
                    this.config.timeout
                )
            )
        ]);
    }
    
    private onSuccess() {
        this.failureCount = 0;
        this.lastFailureTime = null;
        
        if (this.state === "half-open") {
            this.state = "closed";
        }
    }
    
    private onFailure() {
        const now = Date.now();
        
        // Reset counter se fora da janela
        if (
            this.lastFailureTime &&
            now - this.lastFailureTime > this.config.failureWindow
        ) {
            this.failureCount = 1;
        } else {
            this.failureCount++;
        }
        
        this.lastFailureTime = now;
        
        // Abrir se limite de falhas atingido
        if (this.failureCount >= this.config.failureThreshold) {
            this.state = "open";
            this.nextAttemptTime = now + this.config.cooldownMs;
        }
    }
}

export class CircuitBreakerOpenError extends Error {
    constructor(message: string, public retryAfterMs: number) {
        super(message);
        this.name = "CircuitBreakerOpenError";
    }
}
```

## 5. Retry com Backoff Exponencial

### 5.1 Estratégia de Retry

```typescript
// src/lib/retry/retry-strategy.ts

export interface RetryConfig {
    maxAttempts: number;
    initialDelayMs: number;
    maxDelayMs: number;
    backoff: "exponential" | "linear";
    jitter?: boolean;
}

export const retryStrategies = {
    // Integrações rápidas (API calls)
    fast: {
        maxAttempts: 3,
        initialDelayMs: 100,
        maxDelayMs: 1000,
        backoff: "exponential" as const,
        jitter: true
    },
    
    // Operações de longa duração (processamento)
    slow: {
        maxAttempts: 5,
        initialDelayMs: 500,
        maxDelayMs: 10000,
        backoff: "exponential" as const,
        jitter: true
    },
    
    // Operações super críticas
    critical: {
        maxAttempts: 10,
        initialDelayMs: 200,
        maxDelayMs: 30000,
        backoff: "exponential" as const,
        jitter: true
    }
} as const;

export async function withRetry<T>(
    fn: () => Promise<T>,
    config: RetryConfig,
    onRetry?: (attempt: number, error: Error) => void
): Promise<T> {
    let lastError: Error | null = null;
    
    for (let attempt = 1; attempt <= config.maxAttempts; attempt++) {
        try {
            return await fn();
        } catch (error) {
            lastError = error instanceof Error ? error : new Error(String(error));
            
            if (attempt === config.maxAttempts) {
                break; // Última tentativa
            }
            
            // Calcular delay
            let delay = config.initialDelayMs;
            
            if (config.backoff === "exponential") {
                delay = Math.min(
                    config.initialDelayMs * Math.pow(2, attempt - 1),
                    config.maxDelayMs
                );
            }
            
            // Adicionar jitter para evitar thundering herd
            if (config.jitter) {
                delay = delay * (0.5 + Math.random() * 0.5);
            }
            
            onRetry?.(attempt, lastError);
            
            await new Promise(resolve => setTimeout(resolve, delay));
        }
    }
    
    throw lastError ?? new Error("All retry attempts failed");
}
```

```typescript
export async function fetchUserGravatar(email: string): Promise<string | null> {
    try {
        const hash = md5(email.toLowerCase());
```

```typescript
        getRevenue(),
        getUsers()

    return {
        tickets: ticketsResult.status === "fulfilled" ? ticketsResult.value : [],

        users: usersResult.status === "fulfilled" ? usersResult.value : []
    };
}
```

## 7. Monitoramento Proativo

### 7.1 Métricas Críticas

**O que monitorar:**

| Métrica | Threshold | Ação |
|---------|-----------|------|
| Error rate | > 1% | Alert |
| P95 latency | > 500ms | Investigate |
| Circuit breaker open | Qualquer | Alert + fallback |
| Cache miss rate | > 20% | Aumentar TTL |
| Rate limit violations | > 100/hour | Review quota |
| Integração timeout | > 5% | Review timeout config |

### 7.2 Exemplo de Health Check

```typescript
// src/lib/health/health-check.ts

export type HealthStatus = "healthy" | "degraded" | "unhealthy";

interface HealthCheckResult {
    status: HealthStatus;
    timestamp: Date;
    checks: {
        database: { status: boolean; latency: number };
        redis: { status: boolean; latency: number };
        zammad: { status: boolean; latency: number };
    };
}

export async function performHealthCheck(): Promise<HealthCheckResult> {
    const [dbCheck, redisCheck, extCheck] = await Promise.all([
        checkDatabase(),
        checkRedis(),
        checkExternalServices()
    ]);
    
    const allHealthy = [dbCheck, redisCheck, extCheck].every(c => c.status);
    const someFailing = [dbCheck, redisCheck, extCheck].some(c => !c.status);
    
    return {
        status: allHealthy ? "healthy" : someFailing ? "unhealthy" : "degraded",
        timestamp: new Date(),
        checks: {
            database: dbCheck,
            redis: redisCheck,
            zammad: extCheck
        }
    };
}

// API endpoint
export async function GET(req: Request) {
    const health = await performHealthCheck();
    
    return Response.json(health, {
        status: health.status === "healthy" ? 200 : 503
    });
}
```

## Resumo: Checklist de Resiliência

Antes de fazer PR, sua feature deve ter:

- [ ] Tipos de erro estruturados (extends DomainError)
- [ ] Tratamento de erro em actions e no tRPC com `handleActionError`
- [ ] Logging estruturado em JSON com namespace, userId, durationMs
- [ ] Rate limit se aplicável (user/ip/company level)
- [ ] Circuit breaker se chamar integrações externas
- [ ] Retry com exponential backoff para chamadas de rede
- [ ] Graceful degradation onde crítico (fallbacks)
- [ ] Health check se database ou cache crítico
- [ ] Timeout em todas chamadas HTTP/network

Resiliência não é "nice to have" — é essencial para production! 🛡️
